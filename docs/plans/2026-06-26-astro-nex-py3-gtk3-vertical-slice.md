# Astro-Nex Py3 + GTK3 — Vertical Slice Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Port Astro-Nex to Python 3 + GTK 3 with a clean GUI-free core, delivering a vertical slice on macOS Apple Silicon: enter birth data → compute the Huber chart → draw the radix wheel on screen → export PNG and PDF, with calculations verified identical to the original via golden reference data.

**Architecture:** Approach B from the spec — a `core/` package with no GTK/cairo (ephemeris, dates, chart model), a `render/` cairo layer that draws into any `cairo.Context`, `surfaces/` backends (screen/PNG/PDF), and a thin GTK 3 `gui/`. The legacy Python-2 `pysw.py` interface is re-implemented on `pyswisseph`; the rest of the calc code is ported (not rewritten) and pinned to golden charts generated by running the original in Docker once.

**Tech Stack:** Python 3.12+, `pyswisseph`, PyGObject + GTK 3, `pycairo`, `pytest`, `tzdata`/`zoneinfo`. Reference source: the legacy repo (Python 2 / PyGTK 2).

**Spec:** [docs/superpowers/specs/2026-06-26-astro-nex-py3-gtk3-modernization-design.md](../specs/2026-06-26-astro-nex-py3-gtk3-modernization-design.md)

```python
import json, sys
import pysw

EPHEFLAG = 4  # Moshier, matching the legacy default (nex.py ephe_flag = 4)

def dump(ds):
    out = []
    for c in ds:
        jd = pysw.julday(c["y"], c["m"], c["d"], c["h"])
        planets = pysw.planets(jd, EPHEFLAG)        # list of 11 longitudes
        houses = list(pysw.houses(jd, c["lat"], c["lon"]))
```

```python
# -*- coding: utf-8 -*-
"""Swiss Ephemeris access, re-implementing the legacy ``pysw`` interface
on top of pyswisseph. Behavior is pinned to golden data from the original.

Ephemeris flag convention (from legacy nex.py): 4 = Moshier (no .se1 files),
2 = Swiss (when .se1 files are present). 256 = speed.
"""
import swisseph as swe

MOSEPH = 4
SWIEPH = 2
SPEED = 256
KOCH = b"K"

def setpath(path):
    swe.set_ephe_path(str(path))

def delta(jd):
    return swe.deltat(jd)

def julday(y, m, d, h):
    gregflag = swe.JUL_CAL if (y * 10000 + m * 100 + d) < 15821015 else swe.GREG_CAL
    return swe.julday(y, m, d, h, gregflag)

def revjul(jd, gregflag=swe.GREG_CAL):
    return swe.revjul(jd, gregflag)

def _calc(jd, pl, epheflag):
    # pyswisseph raises on error; legacy returned (status, lon, ..., err)
    try:
        xx, retflag = swe.calc(jd, pl, epheflag)
        return retflag, xx, ""
    except swe.Error as exc:  # type: ignore[attr-defined]
        return -1, (0.0,) * 6, str(exc)

def calc(jd, pl, epheflag=MOSEPH):
    # legacy converts UT->ET by adding delta-t before swe_calc
    status, xx, err = _calc(jd + delta(jd), pl, epheflag)
    return status, xx[0], err

def calc_ut(jd, pl, epheflag=MOSEPH):
    # legacy calls swe_calc (ET fn) directly on a UT jd; replicate exactly
    status, xx, err = _calc(jd, pl, epheflag)
    return status, xx[0], err

def calc_ut_with_speed(jd, pl, epheflag=MOSEPH):
    status, xx, err = _calc(jd, pl, epheflag | SPEED)
    return status, xx[0], xx[3], err

def houses(jd, glt, glg):
    cusps, ascmc = swe.houses(jd, glt, glg, KOCH)
    return cusps

def local_houses(jd, glg, glt, epheflag):
    armc = glg + 360 if glg < 0 else glg
    status, eps, err = calc(jd, -1, epheflag)  # -1 = ECL_NUT, obliquity in xx[0]
    cusps, ascmc = swe.houses_armc(armc, glt, eps, KOCH)
    return cusps

def planets(jd, epheflag, p=12):
    out = []
    for i in range(p):
        if i == 10:  # legacy skips index 10 (Earth)
            continue
        status, lon, err = calc(jd, i, epheflag)
        if status < 0:
            return None
        out.append(lon)
    return out
```

```python
# -*- coding: utf-8 -*-
from datetime import tzinfo, timedelta

class FixedOffset(tzinfo):
    """Fixed offset in minutes east of UTC (used for pre-standard-time longitudes)."""
    def __init__(self, minutes):
        if abs(minutes) >= 1440:
            raise ValueError("absolute offset is too large", minutes)
        self._minutes = minutes
        self._offset = timedelta(minutes=minutes)

    def utcoffset(self, dt):
        return self._offset

    def dst(self, dt):
        return timedelta(0)

    def tzname(self, dt):
        return "LMT"

    def __repr__(self):
        return "FixedOffset(%d)" % self._minutes
```

```python
# -*- coding: utf-8 -*-
from datetime import datetime, timedelta
from zoneinfo import ZoneInfo

UTC = ZoneInfo("UTC")

class NeXDate(object):
    """Holds an aware local datetime and its UT projection for ephemeris calc."""
    def __init__(self, dt=None, tz=UTC, current=None):
        self.tz = tz
        self.current = current
        if dt is None:
            dt = datetime.now()
        self.set_local(dt)

    def set_local(self, naive_or_aware):
        dt = naive_or_aware
        if dt.tzinfo is None:
            dt = dt.replace(tzinfo=self.tz)
        self.ld = dt
        self.dt = dt.astimezone(UTC)

    def settz(self, tz):
        self.tz = tz
        self.set_local(self.ld.replace(tzinfo=None))

    def dateforcalc(self):
        t = self.dt
        h = t.hour + (t.minute + t.second / 60.0) / 60.0
        return t.year, t.month, t.day, h
```

```python
        self.set_default_size(1000, 640)
        self.current = None

            left.pack_start(b, False, False, 0)

        self.add(paned)

        self.view.set_chart(chart)
```
